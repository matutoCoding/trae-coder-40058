## 2. 技术描述
- 前端：React@18 + TypeScript + TailwindCSS@3 + Vite
- 初始化工具：vite-init react-ts 模板
- 后端：无（纯前端应用，使用 LocalStorage 持久化 + Mock 数据）
- 数据库：LocalStorage 浏览器本地存储
- 图表库：Recharts
- 图标库：Lucide React
- 状态管理：Zustand

## 4. 数据模型

### 4.1 数据模型定义

```mermaid
erDiagram

        string rubberType "胶种"
        number quantity "数量(kg)"
        string receiver "接收人"
        date receiveDate "接收日期"
        string remark "备注"
    }

    MOLD {
        string id PK "模具ID"
        string moldNo "模具编号"
        string productName "产品名称"
        number preheatTemp "预热温度(℃)"
        number preheatTime "预热时间(min)"
        string status "状态(可用/使用中/维修中)"
        string operator "操作工"
    }

    PLATE_VULCANIZATION {
        string id PK "记录ID"
        string semiFinishedId FK "半成品ID"
        string moldId FK "模具ID"

        number temperature "硫化温度(℃)"
        number duration "硫化时间(min)"
        string temperatureCurve "温度曲线数据"
        string operator "操作工"
        date startTime "开始时间"
        date endTime "结束时间"
    }

    TANK_VULCANIZATION {
        string id PK "记录ID"
        string semiFinishedId FK "半成品ID"
        number steamPressure "蒸汽压力(MPa)"
        number temperature "硫化温度(℃)"
        number duration "硫化时间(min)"
        string operator "操作工"
        date startTime "开始时间"
        date endTime "结束时间"
    }

    DEMOLDING {
        string id PK "记录ID"
        string vulcanizationId FK "硫化记录ID"
        string productNo "产品编号"
        number qualifiedQty "合格品数量"
        number defectiveQty "废品数量"
        string trimmingRecord "毛边修整记录"
        string operator "操作工"
        date createTime "记录时间"
    }

    APPEARANCE_INSPECTION {
        string id PK "记录ID"
        string demoldingId FK "脱模记录ID"
        string bubbleDefect "气泡缺陷记录"
        string lackGlueDefect "缺胶缺陷记录"
        string dimensionCheck "尺寸符合性记录"
        string result "判定结果(合格/不合格)"
        string inspector "质检员"
        date inspectTime "检验时间"
    }

    PHYSICAL_INSPECTION {
        string id PK "记录ID"
        string appearanceId FK "外观检验ID"
        number hardness "邵氏硬度"
        string hardnessResult "硬度判定"
        string agingTestCondition "老化试验条件"
        string agingTestResult "老化试验结果"
        string inspector "质检员"
        date inspectTime "检验时间"
    }

    ENERGY_STATISTICS {
        string id PK "记录ID"
        date statDate "统计日期"
        number electricity "电耗(kWh)"
        number steam "蒸汽耗(t)"
        number water "水耗(t)"
        number totalCost "总能耗费用"
    }
```

## 5. 项目目录结构

```
src/
├── components/          # 公共组件
│   ├── Layout/          # 布局组件
│   ├── Card/            # 统计卡片

│   ├── Form/            # 表单组件
│   └── Chart/           # 图表组件
├── pages/               # 页面组件

└── main.tsx             # 渲染入口
```
